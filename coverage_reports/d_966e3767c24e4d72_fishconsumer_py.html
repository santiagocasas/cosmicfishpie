139def add_mathrm(s): 

140 # The regex pattern will match any sequence of two or more alphanumeric characters 

141 pattern = r"[a-zA-Z0-9]{2,}" 

142 

143 # Find all the substrings that match the pattern 

144 substrings = re.findall(pattern, s) 

145 

146 # Replace the matched substrings with the surrounded string 

147 for substring in substrings: 

148 s = s.replace(substring, r"\mathrm{" + substring + "}") 

149 

150 # Replace spaces with LaTeX proper space '\,' 

151 s = s.replace(" ", r"\,\,") 

152 

153 # Surround the final string with dollar signs 

154 s = f"${s}$" 

155 

156 return s

356def perc_to_abs(perc_sig, fid): 

357 sigma_abs = np.abs(perc_sig / 100 * fid) 

358 return sigma_abs

361def log_fidu_to_fidu(logfid): 

362 fid = np.power(10, logfid) 

363 return fid

366def sigma_fidu(log_fidu, sigma_perc_log, sign): 

367 sigma_log = perc_to_abs(sigma_perc_log, log_fidu) 

368 fidu = log_fidu_to_fidu(log_fidu) 

369 sigma_fid = sign * (np.power(10, log_fidu + sign * sigma_log) - fidu) 

370 return sigma_fid

373def gaussian(x, mu, sigma): 

374 pref = 1 / (sigma * np.sqrt(2 * np.pi)) 

375 expo = np.exp(-0.5 * ((x - mu) / sigma) ** 2) 

376 return pref * expo

379def n_sigmas(log_fidu, sigma_log, nsigmas=1): 

380 fidu = log_fidu_to_fidu(log_fidu) 

381 val_plus = fidu + nsigmas * sigma_fidu(log_fidu, sigma_log, +1) 

382 val_minus = fidu - nsigmas * sigma_fidu(log_fidu, sigma_log, -1) 

383 return [val_minus, val_plus]

386def arrays_gaussian(log_fidu, perc_sigma, nsigmas=1): 

387 sigma_log = perc_to_abs(perc_sigma, log_fidu) 

388 xa = np.linspace(log_fidu - nsigmas * sigma_log, log_fidu + nsigmas * sigma_log, 1000) 

389 g1d = gaussian(xa, log_fidu, sigma_log) 

390 return xa, g1d
